document.addEventListener("DOMContentLoaded", function () {
    const modal = document.getElementById("taskModal");
    const addTaskButton = document.querySelector(".add-task");
    const closeModalButton = document.querySelector(".modal__close");
    const uploadForm = document.getElementById("uploadForm");

    // Funci贸n para abrir el modal
    function openModal() {
        modal.classList.add("show");
    }

    // Funci贸n para cerrar el modal
    function closeModal() {
        modal.classList.remove("show");
    }

    // Evento para abrir el modal al hacer clic en "Subir Archivo"
    addTaskButton.addEventListener("click", openModal);
    
    // Evento para cerrar el modal al hacer clic en el bot贸n de cerrar
    closeModalButton.addEventListener("click", closeModal);
    
    // Cerrar el modal si el usuario hace clic fuera del contenido
    window.addEventListener("click", function (event) {
        if (event.target === modal) {
            closeModal();
        }
    });

    // Evento para manejar la subida de archivos
    uploadForm.addEventListener("submit", function (event) {
        event.preventDefault(); // Evitar el env铆o tradicional del formulario

        const formData = new FormData(uploadForm);

        fetch(uploadForm.action, {
            method: 'POST',
            body: formData
        })
        .then(response => response.json())
        .then(data => {
            if (data.success) {
                showConfirmation('Archivo subido correctamente ');
                closeModal();
                // Recargar la p谩gina o actualizar la lista de archivos
                setTimeout(() => {
                    window.location.reload();
                }, 3000);
            } else {
                showConfirmation('Error al subir el archivo ');
            }
        })
        .catch(error => {
            showConfirmation('Error al subir el archivo ');
        });
    });
});

document.addEventListener("DOMContentLoaded", function () {
    var dropdowns = document.querySelectorAll(".dropdown-toggle");
    dropdowns.forEach(function (dropdown) {
        dropdown.addEventListener("click", function () {
            var dropdownMenu = this.nextElementSibling;
            dropdownMenu.classList.toggle("show");
        });
    });

    // Cerrar el dropdown si se hace clic fuera de 茅l
    window.addEventListener("click", function (event) {
        if (!event.target.matches('.dropdown-toggle')) {
            var dropdowns = document.querySelectorAll(".dropdown-menu");
            dropdowns.forEach(function (dropdownMenu) {
                if (dropdownMenu.classList.contains("show")) {
                    dropdownMenu.classList.remove("show");
                }
            });
        }
    });
});
